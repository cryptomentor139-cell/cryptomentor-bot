Edit app/providers/coinapi.py → ganti fungsi get_ohlcv jadi:

async def get_ohlcv(coin: str, period="5MIN", limit=300, market="spot") -> List[Dict]:
    """
    Selalu coba CoinAPI dulu (symbol_id benar), lalu JIKA GAGAL APA PUN, fallback ke Binance klines.
    """
    try:
        sid = await _resolve_symbol_id(coin, "perp" if market.lower()=="perp" else "spot")
        if sid:
            url = f"{BASE}/ohlcv/{sid}/history"
            params = {"period_id": period, "limit": limit}
            data = await fetch_json(
                url, headers=_headers(), params=params,
                cache_key=f"caohlcv:{sid}:{period}:{limit}", cache_ttl=20
            )
            rows = [{
                "time": d.get("time_period_end"),
                "open": float(d["price_open"]),
                "high": float(d["price_high"]),
                "low":  float(d["price_low"]),
                "close":float(d["price_close"]),
                "volume": float(d.get("volume_traded", 0.0)),
            } for d in data]
            if rows:
                return rows
    except Exception as e:
        # not raising — we will fallback below
        try:
            # optional: simple stderr print (agar kebaca di logs Replit)
            import sys
            print(f"[CoinAPI OHLCV ERROR] {coin} market={market}: {e}", file=sys.stderr)
        except Exception:
            pass

    # Fallback kuat → Binance public klines (USDT pair)
    from app.providers.binance import get_klines as binance_klines
    symbol = f"{_normalize_coin(coin)}USDT"
    return await binance_klines(symbol=symbol, interval="5m", limit=limit)


Perbedaan penting: seluruh panggilan CoinAPI kini dibungkus try/except. Jika error, tidak melempar—langsung fallback.

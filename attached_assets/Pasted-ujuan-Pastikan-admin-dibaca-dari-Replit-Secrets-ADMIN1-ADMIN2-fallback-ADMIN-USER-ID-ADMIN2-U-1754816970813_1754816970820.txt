# app/lib/auth.py
import os

def _resolve_admin_ids():
    # Primary
    a1 = (os.getenv("ADMIN1") or "").strip().strip('"').strip("'")
    a2 = (os.getenv("ADMIN2") or "").strip().strip('"').strip("'")
    # Fallback legacy
    if not a1:
        a1 = (os.getenv("ADMIN_USER_ID") or "").strip().strip('"').strip("'")
    if not a2:
        a2 = (os.getenv("ADMIN2_USER_ID") or "").strip().strip('"').strip("'")
    # Normalisasi & buang kosong
    ids = {s for s in (str(a1), str(a2)) if s and s.lower() != "none"}
    return ids

def is_admin(user_id) -> bool:
    if user_id is None:
        return False
    return str(user_id) in _resolve_admin_ids()

2) Perbaiki dekorator guard supaya aman untuk method/function
Buat/overwrite app/lib/guards.py:

python
Copy
# app/lib/guards.py
from functools import wraps
from app.lib.auth import is_admin

def admin_guard(func):
    @wraps(func)
    async def _inner(*args, **kwargs):
        # Deteksi method (self, update, context) vs function (update, context)
        if len(args) >= 3 and not hasattr(args[0], "effective_user"):
            self_obj, update, context = args[0], args[1], args[2]
            tail = args[3:]
        elif len(args) >= 2:
            self_obj = None
            update, context = args[0], args[1]
            tail = args[2:]
        else:
            raise RuntimeError("Invalid handler signature")

        # Ambil UID aman
        uid = getattr(getattr(update, "effective_user", None), "id", None)
        if not is_admin(uid):
            msg = getattr(update, "effective_message", None)
            if msg:
                await msg.reply_text("❌ Akses ditolak. Command ini hanya untuk admin.")
            return

        if self_obj is not None:
            return await func(self_obj, update, context, *tail, **kwargs)
        return await func(update, context, *tail, **kwargs)
    return _inner

3) Tambah debug commands
Buat/overwrite app/handlers_admin_debug.py:

# app/handlers_admin_debug.py
from telegram import Update
from telegram.ext import ContextTypes
from app.lib.auth import is_admin, _resolve_admin_ids

async def cmd_whoami(update: Update, context: ContextTypes.DEFAULT_TYPE):
    uid = getattr(update.effective_user, "id", None)
    await update.effective_message.reply_text(f"👤 Your ID: {uid}")

async def cmd_admin_debug(update: Update, context: ContextTypes.DEFAULT_TYPE):
    uid = getattr(update.effective_user, "id", None)
    ids = sorted(list(_resolve_admin_ids()))
    await update.effective_message.reply_text(
        "🔧 Admin Debug\n"
        f"- caller_id: {uid}\n"
        f"- resolved_admin_ids: {ids}\n"
        f"- is_admin(caller): {is_admin(uid)}"
    )

--- app/middlewares/ensure_weekly_reset.py (FILE BARU, aiogram v3) ---
from aiogram import BaseMiddleware
from aiogram.types import TelegramObject, Message, CallbackQuery
from typing import Callable, Dict, Any, Awaitable
from ..sb_client import upsert_user_with_weekly_reset_rpc, enforce_weekly_reset_calendar_rpc

def _touch(obj):
    u = getattr(obj, "from_user", None)
    if not u: return
    # 1) pastikan user terdaftar + user baru = 100 credits
    upsert_user_with_weekly_reset_rpc(
        u.id,
        getattr(u, "username", None),
        getattr(u, "first_name", None),
        getattr(u, "last_name", None),
    )
    # 2) enforce reset mingguan (Sen 00:00 UTC), idempotent utk non-premium
    try:
        enforce_weekly_reset_calendar_rpc(u.id)
    except Exception:
        pass

class EnsureWeeklyResetMiddleware(BaseMiddleware):
    async def __call__(self,
                       handler: Callable[[TelegramObject, Dict[str, Any]], Awaitable[Any]],
                       event: TelegramObject,
                       data: Dict[str, Any]) -> Any:
        if isinstance(event, Message):
            _touch(event)
        elif isinstance(event, CallbackQuery):
            _touch(event)
        return await handler(event, data)

Tujuan

AutoSignal hanya mengirim ke:

Admin (ADMIN1, ADMIN2 … fallback ADMIN_USER_ID, ADMIN2_USER_ID)

Lifetime premium (Supabase: is_premium=true & premium_until IS NULL; Legacy JSON: lifetime=true / premium_until kosong)

Tetap hormati banned (exclude) dan DM consent (user sudah /start privat).

Tambah command admin /signal_audience untuk cek daftar penerima saat ini.

# app/users_aggregate.py (tambahkan import typing & datetime sudah ada di file)
def list_supabase_lifetime(limit: int = 2000, offset: int = 0) -> Dict[int, Dict[str, Any]]:
    out: Dict[int, Dict[str, Any]] = {}
    rows = sb_list_users({
        "is_premium": "eq.true",
        "banned": "eq.false",
        "premium_until": "is.null",
        "limit": str(limit),
        "offset": str(offset),
    })
    for r in rows:
        tid = r.get("telegram_id")
        if tid is None:
            continue
        out[int(tid)] = {
            "telegram_id": int(tid),
            "is_premium": True,
            "premium_until": None,   # lifetime
            "banned": False,
            "source": "supabase",
            "credits": int(r.get("credits", 0) or 0),
        }
    return out

def list_old_lifetime() -> Dict[int, Dict[str, Any]]:
    out: Dict[int, Dict[str, Any]] = {}
    for r in _load_old_backup():
        m = _normalize_old_record(r)
        if not m:
            continue
        if m["banned"]:
            continue
        # lifetime kalau premium_until None/"" ATAU flag lifetime ada
        if (m.get("is_premium") and (m.get("premium_until") in (None, "", "null"))) or \
           (str(r.get("plan","")).lower().find("lifetime") >= 0) or bool(r.get("lifetime")):
            out[m["telegram_id"]] = {
                **m,
                "is_premium": True,
                "premium_until": None,
                "source": "old",
            }
    return out

def aggregated_lifetime() -> Dict[int, Dict[str, Any]]:
    """Merge lifetime dari Supabase + Old backup + (opsional) local runtime (jika ada & lifetime)."""
    sup = list_supabase_lifetime()
    old = list_old_lifetime()

    # Optional: lifetime dari storage_local
    loc = {}
    try:
        users = load_users()
        for k, v in users.items():
            try: tid = int(k)
            except: continue
            if v.get("banned"): continue
            if bool(v.get("is_premium")) and v.get("premium_until") in (None, "", "null"):
                loc[tid] = {
                    "telegram_id": tid,
                    "is_premium": True,
                    "premium_until": None,
                    "banned": False,
                    "source": "local",
                    "credits": int(v.get("credits", 0) or 0),
                }
    except Exception:
        pass

    merged: Dict[int, Dict[str, Any]] = {}
    def _merge(dst, src):
        if not dst: return src.copy()
        # credits keep max
        dst["credits"] = max(int(dst.get("credits",0)), int(src.get("credits",0)))
        dst["source"] = "merge"
        return dst

    for d in (old, loc, sup):
        for tid, rec in d.items():
            merged[tid] = _merge(merged.get(tid, {}), rec)

    return merged

# app/autosignal.py
from app.users_aggregate import aggregated_lifetime
from app.chat_store import get_private_chat_id
from app.lib.auth import _resolve_admin_ids  # fungsi yang baca ADMIN1/2 & fallback

def _admin_ids_as_int() -> list[int]:
    ids = []
    for s in _resolve_admin_ids():
        try: ids.append(int(s))
        except: pass
    return ids

def list_recipients() -> List[int]:
    """
    Hanya Admin + Lifetime users. Exclude yg tidak punya private chat consent.
    """
    recipients: set[int] = set()

    # Admins (boleh tanpa consent? default: tetap butuh consent agar tidak 403)
    for aid in _admin_ids_as_int():
        if get_private_chat_id(aid) is not None:
            recipients.add(aid)

    # Lifetime from merged sources
    lifetime = aggregated_lifetime()  # dict {tid -> record}
    for tid in lifetime.keys():
        if get_private_chat_id(int(tid)) is not None:
            recipients.add(int(tid))

    return sorted(list(recipients))
Kalau kamu mau admin selalu menerima meski belum /start privat, ganti blok admin dengan langsung recipients.add(aid)—tapi Telegram bisa balas 403 kalau admin belum /start DM. Rekomendasi tetap pakai consent check.

# app/handlers_signal_audience.py
from telegram import Update
from telegram.ext import ContextTypes
from app.lib.guards import admin_guard
from app.autosignal import list_recipients, _admin_ids_as_int
from app.users_aggregate import aggregated_lifetime
from app.safe_send import safe_reply

@admin_guard
async def cmd_signal_audience(update: Update, context: ContextTypes.DEFAULT_TYPE):
    recs = list_recipients()
    lif = aggregated_lifetime()
    admins = _admin_ids_as_int()
    lines = [
        "🛰️ AutoSignal Audience",
        f"- admins: {admins}",
        f"- lifetime_count: {len(lif)}",
        f"- recipients_total: {len(recs)}",
        f"- sample: {recs[:20]}",
    ]
    await safe_reply(update.effective_message, "\n".join(lines))

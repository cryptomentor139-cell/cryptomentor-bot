from __future__ import annotations
from typing import Dict, Any, List, Tuple
import math
from datetime import datetime

from app.providers.binance import get_top_usdt_coins
from app.providers.http import fetch_json

BINANCE_URL = "https://api.binance.com"

async def _get_24h_all() -> List[Dict[str, Any]]:
    url = f"{BINANCE_URL}/api/v3/ticker/24hr"
    return await fetch_json(url, timeout=12, cache_key="bn:24hr", cache_ttl=15)

def _classify_sentiment(breadth_pct: float) -> str:
    # breadth = % coin hijau 24h
    if breadth_pct >= 60:
        return "Bullish"
    if breadth_pct <= 40:
        return "Bearish"
    return "Neutral"

async def get_market_sentiment(top_n: int = 20) -> Dict[str, Any]:
    """
    Ambil top_n pair USDT dengan volume 24h terbesar dari Binance,
    hitung breadth (persen koin hijau hari ini) sebagai proxy sentimen.
    """
    top = await get_top_usdt_coins(limit=top_n)
    all24 = await _get_24h_all()

    by_symbol = {d.get("symbol"): d for d in all24}
    scanned: List[Dict[str, Any]] = []
    green = 0
    total = 0

    for coin in top:
        sym = f"{coin}USDT"
        d = by_symbol.get(sym)
        if not d:
            continue
        total += 1
        try:
            change_pct = float(d.get("priceChangePercent", 0.0))
            last_price = float(d.get("lastPrice", 0.0))
            qv = float(d.get("quoteVolume", 0.0))
        except Exception:
            change_pct, last_price, qv = 0.0, 0.0, 0.0
        if change_pct > 0:
            green += 1
        scanned.append({
            "coin": coin,
            "price": last_price,
            "change_24h": change_pct,
            "quote_volume": qv,
        })

    breadth_pct = (green / total * 100.0) if total else 0.0
    sentiment = _classify_sentiment(breadth_pct)

    scanned.sort(key=lambda x: abs(x.get("change_24h", 0.0)), reverse=True)

    return {
        "time": datetime.now().strftime("%d-%m-%Y %H:%M:%S WIB"),
        "universe": f"Top {top_n} USDT by 24h volume",
        "breadth_pct": breadth_pct,
        "sentiment": sentiment,
        "coins": scanned,   # list of {coin, price, change_24h, quote_volume}
        "counts": {"green": green, "red": total - green, "total": total},
    }

from app.utils.telegram_safe import safe_reply  # jika belum ada, pakai reply_text biasa
from app.services.market import get_market_sentiment

def _fmt_price(v: float) -> str:
    try:
        return f"${v:,.4f}".replace(",", "X").replace(".", ",").replace("X", ".")  # jika mau locale, opsional
    except Exception:
        return str(v)

async def cmd_market(update, context):
    try:
        # argumen opsional: jumlah top coin (default 20)
        try:
            top_n = int(context.args[0]) if context.args else 20
            top_n = max(5, min(50, top_n))  # batasi agar tidak berat
        except Exception:
            top_n = 20

        res = await get_market_sentiment(top_n=top_n)

        header = (
            "🌐 <b>Market Overview</b>\n"
            f"🕐 <i>{res['time']}</i>\n"
            f"📈 Universe: {res['universe']}\n"
            f"🟢 Breadth: <b>{res['breadth_pct']:.2f}%</b> → <b>{res['sentiment']}</b>\n"
            f"🧮 Green/Red/Total: {res['counts']['green']}/{res['counts']['red']}/{res['counts']['total']}\n"
        )

        # Tampilkan 10 koin ter-bergerak (abs change paling besar)
        lines = ["\n🔎 Movers (Top 10 by |Δ|):"]
        for it in res["coins"][:10]:
            sign = "🟢" if it["change_24h"] > 0 else "🔴" if it["change_24h"] < 0 else "⚪"
            lines.append(
                f"{sign} {it['coin']}: {_fmt_price(it['price'])} ({it['change_24h']:.2f}%)"
            )

        text = header + "\n".join(lines)
        await safe_reply(update, text, prefer_html=True)
    except Exception as e:
        err = (
            "❌ Terjadi kesalahan saat menganalisis pasar.\n\n"
            f"Error: {e}\n\n"
            "💡 Coba alternatif:\n"
            "• /price btc (CoinAPI)\n"
            "• /analyze ethereum (CoinAPI)\n"
            "• /futures_signals (SnD)\n"
        )
        await safe_reply(update, err, prefer_html=True)

def _format_md_v2(text: str) -> str:
    # Escape semua karakter spesial MarkdownV2:
    # _ * [ ] ( ) ~ ` > # + - = | { } . !
    return escape_markdown(text, version=2)

async def _safe_reply_status(message, raw_text: str):
    """
    Coba kirim sebagai MarkdownV2 (escaped).
    Jika gagal, fallback ke HTML <pre> (escaped).
    Jika masih gagal, kirim plain text tanpa parse mode.
    """
    # 1) MarkdownV2 (escaped)
    try:
        md2 = _format_md_v2(raw_text)
        await message.reply_text(md2, parse_mode=ParseMode.MARKDOWN_V2)
        return
    except Exception as e_md2:
        pass

    # 2) HTML <pre> (escaped)
    try:
        html_text = f"<pre>{html.escape(raw_text)}</pre>"
        await message.reply_text(html_text, parse_mode=ParseMode.HTML)
        return
    except Exception as e_html:
        pass

    # 3) Plain text
    await message.reply_text(raw_text)

Kenapa ini bekerja
escape_markdown(..., version=2) memastikan setiap karakter spesial MarkdownV2 di-escape, jadi teks seperti public.users, root_ok, SUPABASE_SERVICE_KEY aman.

Fallback HTML <pre> + html.escape() menjaga JSON/trace panjang tetap rapi.

Jika Telegram API masih rewel, kita kirim plain text tanpa parse mode sebagai jaring pengaman terakhir.

Coba variabel dengan underscore (SUPABASE_SERVICE_KEY) → tidak error lagi.

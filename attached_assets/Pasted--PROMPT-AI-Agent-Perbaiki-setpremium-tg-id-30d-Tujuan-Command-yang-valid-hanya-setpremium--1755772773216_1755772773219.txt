🔧 PROMPT AI Agent — Perbaiki /setpremium <tg_id> 30d

Tujuan

Command yang valid hanya /setpremium <tg_id> <duration> (+ alias /revoke_premium <tg_id>).

30d / 30 → (days, 30), 2m → (months, 2), lifetime → lifetime.

Pastikan user dibuat dulu meski belum /start.

Panggil RPC set_premium(p_telegram_id, p_duration_value, p_duration_type) dengan argumen yang dinormalisasi.

Verifikasi dengan membaca ulang dari v_users dan tampilkan premium_active.

1) Repo Supabase (normalisasi di level repo)

File: app/supabase_repo.py
Tambahkan/replace fungsi berikut:

# app/supabase_repo.py
import os
from typing import Optional, Dict, Any, Tuple
from supabase import create_client, Client

SUPABASE_URL = (os.getenv("SUPABASE_URL") or "").rstrip("/")
SUPABASE_SERVICE_KEY = os.getenv("SUPABASE_SERVICE_KEY")

def _client() -> Client:
    if not SUPABASE_URL or not SUPABASE_SERVICE_KEY:
        raise RuntimeError("Set SUPABASE_URL & SUPABASE_SERVICE_KEY (Service role).")
    return create_client(SUPABASE_URL, SUPABASE_SERVICE_KEY)

def get_user_by_tid(tg_id: int) -> Optional[Dict[str, Any]]:
    s = _client()
    res = s.table("users").select("*").eq("telegram_id", int(tg_id)).limit(1).execute()
    return res.data[0] if res.data else None

def get_vuser_by_tid(tg_id: int) -> Optional[Dict[str, Any]]:
    s = _client()
    res = s.table("v_users").select("*").eq("telegram_id", int(tg_id)).limit(1).execute()
    return res.data[0] if res.data else None

def ensure_user_exists(
    tg_id: int,
    username: Optional[str] = None,
    first_name: Optional[str] = None,
    last_name: Optional[str] = None,
) -> Dict[str, Any]:
    s = _client()
    # Coba RPC upsert_user_with_welcome (welcome=0 agar tidak menyentuh kredit)
    try:
        payload = {
            "p_telegram_id": int(tg_id),
            "p_username": (username or "").strip().lstrip("@").lower() or None,
            "p_first_name": first_name,
            "p_last_name": last_name,
            "p_welcome_quota": 0,
            "p_referred_by": None,
        }
        data = s.rpc("upsert_user_with_welcome", payload).execute().data
        if data:
            return data
    except Exception:
        pass
    # Fallback upsert langsung
    row = {
        "telegram_id": int(tg_id),
        "username": (username or "").strip().lstrip("@").lower() or None,
        "first_name": first_name,
        "last_name": last_name,
    }
    data = s.table("users").upsert(row, on_conflict="telegram_id").execute().data
    if data: return data[0]
    # Ambil ulang bila tidak ada return
    r2 = s.table("users").select("*").eq("telegram_id", int(tg_id)).limit(1).execute()
    if r2.data: return r2.data[0]
    raise RuntimeError("ensure_user_exists failed")

def _normalize_duration(token: str) -> Tuple[str, int]:
    """
    lifetime         -> ('lifetime', 0)
    30d / 30         -> ('days', 30)
    2m / 2mo / 2mon  -> ('months', 2)
    """
    t = (token or "").strip().lower()
    if t == "lifetime":
        return ("lifetime", 0)
    if t.isdigit():
        return ("days", int(t))
    if t.endswith("d") and t[:-1].isdigit():
        return ("days", int(t[:-1]))
    if t.endswith(("m", "mo", "mon", "month", "months")):
        num = "".join(ch for ch in t if ch.isdigit())
        if num.isdigit():
            return ("months", int(num))
    raise ValueError("Invalid duration. Use lifetime | <days>d | <days> | <months>m")

def set_premium_normalized(tg_id: int, duration_token: str) -> Dict[str, Any]:
    """Terima '30d'/'30'/'2m'/'lifetime', kirim param valid ke RPC, lalu baca ulang dari v_users."""
    s = _client()
    dtype, dval = _normalize_duration(duration_token)
    s.rpc("set_premium", {
        "p_telegram_id": int(tg_id),
        "p_duration_value": int(dval),
        "p_duration_type": dtype,  # 'days' | 'months' | 'lifetime'
    }).execute()
    # verifikasi dari view
    v = get_vuser_by_tid(tg_id)
    if not v: 
        raise RuntimeError("Premium updated but v_users row not found")
    return v

def revoke_premium(tg_id: int) -> Dict[str, Any]:
    s = _client()
    s.table("users").update({
        "is_premium": False,
        "is_lifetime": False,
        "premium_until": None
    }).eq("telegram_id", int(tg_id)).execute()
    v = get_vuser_by_tid(tg_id)
    if not v: 
        raise RuntimeError("Revoke ok but v_users row not found")
    return v

2) Router admin: /setpremium (dan revoke)

File: app/routers/admin_premium.py
Ganti/buat handler berikut (commandnya /setpremium):

# app/routers/admin_premium.py
import os
from aiogram import Router, types
from app.supabase_repo import ensure_user_exists, set_premium_normalized, revoke_premium

router = Router()

def _admin_ids() -> set[int]:
    ids = set()
    for key in ("ADMIN_IDS", "ADMIN1", "ADMIN2"):
        raw = os.getenv(key, "")
        if not raw: 
            continue
        for tok in raw.replace(";", ",").replace(" ", "").split(","):
            if tok.isdigit():
                ids.add(int(tok))
    return ids

def _is_admin(uid: int) -> bool:
    return uid in _admin_ids()

@router.message(commands={"setpremium"})
async def setpremium_cmd(m: types.Message):
    uid = m.from_user.id if m.from_user else 0
    if not _is_admin(uid):
        return await m.answer(f"❌ Admin only. Your ID: {uid}")

    parts = (m.text or "").split()
    if len(parts) < 3:
        return await m.answer("Usage: /setpremium <telegram_id> <lifetime|<days>d|<days>|<months>m>")

    try:
        tg_id = int(parts[1])
    except Exception:
        return await m.answer("ID tidak valid.")

    dur = parts[2]
    try:
        # Pastikan user ada walau belum /start
        ensure_user_exists(tg_id)
        v = set_premium_normalized(tg_id, dur)
        await m.answer(
            "✅ Premium updated\n"
            f"user={tg_id}\n"
            f"premium={v.get('is_premium')} lifetime={v.get('is_lifetime')} active={v.get('premium_active')}\n"
            f"until={v.get('premium_until')}"
        )
    except Exception as e:
        await m.answer(f"⚠️ setpremium failed: {e}")

@router.message(commands={"revoke_premium"})
async def revoke_cmd(m: types.Message):
    uid = m.from_user.id if m.from_user else 0
    if not _is_admin(uid):
        return await m.answer(f"❌ Admin only. Your ID: {uid}")

    parts = (m.text or "").split()
    if len(parts) < 2:
        return await m.answer("Usage: /revoke_premium <telegram_id>")
    try:
        tg_id = int(parts[1])
    except Exception:
        return await m.answer("ID tidak valid.")

    try:
        v = revoke_premium(tg_id)
        await m.answer(
            "✅ Premium revoked\n"
            f"user={tg_id}\n"
            f"premium={v.get('is_premium')} lifetime={v.get('is_lifetime')} active={v.get('premium_active')}\n"
            f"until={v.get('premium_until')}"
        )
    except Exception as e:
        await m.answer(f"⚠️ revoke_premium failed: {e}")

4) Uji cepat

/setpremium <id> lifetime → is_premium=true, is_lifetime=true, premium_active=true, until=null.

/setpremium <id> 30d atau /setpremium <id> 30 → is_premium=true, is_lifetime=false, premium_until = now() + 30 days, premium_active=true.

/setpremium <id> 2m → premium_until = now() + 2 months.

/revoke_premium <id> → semua flag kembali false / null.

/whois <id> menampilkan status dari v_users (sumber kebenaran).
